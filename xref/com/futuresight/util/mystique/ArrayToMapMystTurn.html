10  package com.futuresight.util.mystique;
11  
12  import java.util.List;
13  
14  import org.springframework.stereotype.Component;
15  
16  import com.futuresight.util.mystique.lever.MysCon;
17  import com.google.gson.JsonArray;
18  import com.google.gson.JsonElement;
19  import com.google.gson.JsonObject;
20  
21  /**
22   * The Class ArrayToMapMystTurn.
23   *
24   * @author balajeetm
25   */
26  @Component
27  public class ArrayToMapMystTurn extends AbstractMystTurn {
28  
29  	/* (non-Javadoc)
30  	 * @see com.futuresight.util.mystique.AbstractMystique#transmute(java.util.List, com.google.gson.JsonObject, com.google.gson.JsonObject)
31  	 */
32  	@Override
33  	protected JsonElement transmute(List<JsonElement> source, JsonObject deps, JsonObject aces, JsonObject turn) {
34  		JsonObject mapJson = new JsonObject();
35  		JsonElement elementSource = jsonLever.getFirst(source);
36  
37  		if (null != elementSource) {
38  			turn = jsonLever.getAsJsonObject(turn, new JsonObject());
39  			JsonElement granularSource = getGranularSource(elementSource, turn, deps, aces);
40  			JsonArray inputArray = jsonLever.getAsJsonArray(granularSource, new JsonArray());
41  			JsonArray keyArray = jsonLever.getAsJsonArray(turn.get(MysCon.KEY));
42  			if (jsonLever.isNotNull(keyArray)) {
43  				JsonElement valueElement = turn.get(MysCon.VALUE);
44  				valueElement = jsonLever.isNull(valueElement) ? new JsonArray() : valueElement;
45  
46  				for (JsonElement jsonElement : inputArray) {
47  					JsonElement keyField = jsonLever.getField(jsonElement, keyArray, deps, aces);
48  					String key = jsonLever.getAsString(keyField, MysCon.EMPTY);
49  
50  					JsonElement finalValue = jsonLever.getSubset(jsonElement, deps, aces, valueElement);
51  					mapJson.add(key, finalValue);
52  				}
53  			}
54  		}
55  
56  		return mapJson;
57  	}
58  }
