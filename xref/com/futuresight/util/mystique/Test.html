17  		public I func;

20  	static Function<Integer, Double> factorial = x -> {
21  		Recursive<IntToDoubleFunction> recursive = new Recursive<IntToDoubleFunction>();
22  		recursive.func = n -> (n == 0) ? 1 : n * recursive.func.applyAsDouble(n - 1);
23  
24  		return recursive.func.applyAsDouble(x);
25  	};

29  	public static void main1(String[] args) throws InterruptedException, ExecutionException {
30  		System.out.println(Thread.currentThread().getName() + " thread enters main method");
31  
32  		ExecutorService newFixedThreadPool = Executors.newFixedThreadPool(5);
33  
34  		final int nthFactorial = 5;
35  
36  		// 1 CompletableFuture stage for calculating factorial
37  		CompletableFuture<Double> factorialCalcStage = CompletableFuture.supplyAsync(() -> {
38  			System.out.println(Thread.currentThread().getName() + " factorial task is called");
39  			Double factorialResult = factorial.apply(nthFactorial);
40  			return factorialResult;
41  		}, newFixedThreadPool);
42  
43  		// 2 creates another CompletableFuture.
44  		factorialCalcStage.thenAcceptAsync(t -> System.out.println(Thread.currentThread().getName()
45  				+ " The result after factorial: " + t));
46  
47  		// 3 creates another CompletableFuture for squaring .
48  		CompletableFuture<Double> squareCalcStage = factorialCalcStage.thenApplyAsync(r -> {
49  			System.out.println(Thread.currentThread().getName() + " square task is called");
50  			Double squareResult = square.apply(r);
51  
52  			return squareResult;
53  		}, newFixedThreadPool);
54  
55  		// 4 creates another CompletableFuture.
56  		CompletableFuture<Void> lastStage = squareCalcStage.thenAcceptAsync(t -> System.out.println(Thread
57  				.currentThread().getName() + " The result after square: " + t));
58  
59  		lastStage.join(); //waits for the lastStage to complete
60  
61  		newFixedThreadPool.shutdown();
62  
63  	}

77  				sleep(5);
78  				return new ArrayList<String>(Arrays.asList("English, Maths, Science"));
79  			});
80  
81  		//3 combine //1 & //2 to produce a report synchronously
82  		final CompletableFuture<String> report = users.thenCombine(subjects, (u, s) -> {
83  			System.out.println("Thread " + Thread.currentThread().getName() + " is producing a report'" + "");
84  			return u.toString() + s.toString();
85  		});
