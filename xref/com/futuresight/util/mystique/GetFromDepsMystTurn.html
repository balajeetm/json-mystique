10  package com.futuresight.util.mystique;
11  
12  import java.util.List;
13  
14  import org.springframework.stereotype.Component;
15  
16  import com.futuresight.util.mystique.lever.MysCon;
17  import com.google.gson.JsonArray;
18  import com.google.gson.JsonElement;
19  import com.google.gson.JsonNull;
20  import com.google.gson.JsonObject;
21  
22  /**
23   * The Class GetFromDepsMystTurn.
24   *
25   * @author balajeetm
26   */
27  @Component
28  public class GetFromDepsMystTurn extends AbstractMystTurn {
29  
30  	/* (non-Javadoc)
31  	 * @see com.futuresight.util.mystique.AbstractMystique#transmute(java.util.List, com.google.gson.JsonObject, com.google.gson.JsonObject)
32  	 */
33  	@Override
34  	protected JsonElement transmute(List<JsonElement> source, JsonObject deps, JsonObject aces, JsonObject turn) {
35  
36  		JsonElement transform = JsonNull.INSTANCE;
37  		JsonElement elementSource = jsonLever.getFirst(source);
38  		if (null != elementSource) {
39  			//element source is key
40  
41  			turn = jsonLever.getAsJsonObject(turn, new JsonObject());
42  			JsonElement granularSource = getGranularSource(elementSource, turn, deps, aces);
43  			String reference = jsonLever.getAsString(granularSource, null);
44  
45  			JsonArray keyPath = jsonLever.getAsJsonArray(turn.get(MysCon.KEY), new JsonArray());
46  			JsonElement value = turn.get(MysCon.VALUE);
47  			value = jsonLever.isNull(value) ? new JsonArray() : value;
48  
49  			//keymap - The source is deps
50  			JsonObject keyMap = jsonLever.getAsJsonObject(jsonLever.getField(deps, keyPath, deps, aces),
51  					new JsonObject());
52  
53  			JsonElement actualElement = keyMap.get(reference);
54  
55  			transform = jsonLever.getSubset(actualElement, deps, aces, value);
56  		}
57  		return transform;
58  	}
59  
60  }
