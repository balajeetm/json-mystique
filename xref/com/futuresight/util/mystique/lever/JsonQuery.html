24  import com.futuresight.util.mystique.JsonLever;
25  import com.google.gson.JsonArray;

31  /**
32   * The Class JsonQuery.
33   *
34   * @author balajeetm
35   */
36  @Component
37  public class JsonQuery {
38  
39  	/** The json lever. */
40  	@Autowired
41  	private JsonLever jsonLever;
42  
43  	/** The query types. */
44  	private Map<String, BiFunction<JsonElement, JsonObject, Boolean>> queryTypes;
45  
46  	/**
47  	 * Instantiates a new json query.
48  	 */
49  	private JsonQuery() {
50  		queryTypes = new HashMap<>();
51  		queryTypes.put("=", this::equalsFilter);
52  		queryTypes.put("!=", this::notEqualsFilter);
53  		queryTypes.put("in", this::inFilter);
54  		queryTypes.put("notin", this::notinFilter);
55  	}
56  
57  	/**
58  	 * Query.
59  	 *
60  	 * @param query the query
61  	 * @return the json element
62  	 */
63  	public JsonElement query(JsonObject query) {
64  
65  		JsonElement result = null;
66  		try {
67  			result = queryAsync(query).reduce(new JsonArray(), (resultarray, json) -> {
68  				resultarray.add(json);
69  				return resultarray;
70  			}).toBlocking().first();
71  		}
72  		catch (RuntimeException e) {
73  			result = new JsonPrimitive(e.getMessage());
74  		}
75  
76  		return result;
77  	}
78  
79  	/**
80  	 * Query async.
81  	 *
82  	 * @param query the query
83  	 * @return the observable
84  	 */
85  	public Observable<JsonElement> queryAsync(JsonObject query) {
86  		return Observable
87  				.just(query)
88  				.map(q -> validate(q))
89  				.flatMap(
90  						s -> {
91  							Observable<JsonElement> obs = null;
92  							if (null == s) {
93  								JsonElement from = query.get("from");
94  								obs = jsonLever.isJsonArray(from) ? queryAsync(from.getAsJsonArray(), query)
95  										: queryAsync(from.getAsJsonObject(), query);
96  							}
97  							else {
98  								Exceptions.propagate(new Throwable(s));
99  							}
100 							return obs;
101 						});
102 	}
103 
104 	/**
105 	 * Query async.
106 	 *
107 	 * @param array the array
108 	 * @param query the query
109 	 * @return the observable
110 	 */
111 	private Observable<JsonElement> queryAsync(JsonArray array, JsonObject query) {
112 		return Observable
113 				.from(array)
114 				.observeOn(Schedulers.computation())
115 				.filter(json -> filter(json, query.get("where")))
116 				.compose(obs -> {
117 					Long limit = jsonLever.getAsLong(query.get("limit"), 0L);
118 					return limit > 0 ? obs.limit(limit.intValue()) : obs;
119 				})
120 				.observeOn(Schedulers.computation())
121 				.compose(
122 						obs -> StringUtils.equals(jsonLever.getFieldAsString(query, "select"), "count") ? obs.count()
123 								.map(count -> new JsonPrimitive(count)) : obs.map(json -> select(json,
124 								query.get("select"))));
125 	}

127 	/**
128 	 * Query async.
129 	 *
130 	 * @param obj the obj
131 	 * @param query the query
132 	 * @return the observable
133 	 */
134 	private Observable<JsonElement> queryAsync(JsonElement obj, JsonObject query) {
135 		return Observable
136 				.just(obj)
137 				.observeOn(Schedulers.computation())
138 				.filter(json -> filter(json, query.get("where")))
139 				.compose(obs -> {
140 					Long limit = jsonLever.getAsLong(query.get("limit"), 0L);
141 					return limit > 0 ? obs.limit(limit.intValue()) : obs;
142 				})
143 				.observeOn(Schedulers.computation())
144 				.compose(
145 						obs -> StringUtils.equals(jsonLever.getFieldAsString(query, "select"), "count") ? obs.count()
146 								.map(count -> new JsonPrimitive(count)) : obs.map(json -> select(json,
147 								query.get("select"))));
148 	}

150 	/**
151 	 * Select.
152 	 *
153 	 * @param json the json
154 	 * @param selectElement the select element
155 	 * @return the json element
156 	 */
157 	private JsonElement select(JsonElement json, JsonElement selectElement) {
158 		JsonElement result = null;
159 		String selectStr = jsonLever.getAsString(selectElement);
160 		if (null != selectStr) {
161 			if (equalsAny(selectStr, "*"))
162 				result = json;
163 		}
164 		else {
165 			if (jsonLever.isJsonArray(selectElement)) {
166 				result = jsonLever.getSubset(json, selectElement);
167 			}
168 		}
169 		return result;
170 	}
171 
172 	/**
173 	 * Filter.
174 	 *
175 	 * @param json the json
176 	 * @param whereElement the where element
177 	 * @return the boolean
178 	 */
179 	private Boolean filter(JsonElement json, JsonElement whereElement) {
180 		Boolean filter = Boolean.TRUE;
181 		JsonArray where = jsonLever.getAsJsonArray(whereElement, new JsonArray());
182 		for (JsonElement condition : where) {
183 			if (jsonLever.isString(condition)) {
184 				String operator = StringUtils.lowerCase(jsonLever.getAsString(condition));
185 				if (equalsAny(operator, "&", "&&", "and")) {
186 					if (filter) {
187 						continue;
188 					}
189 					else {
190 						return filter;
191 					}
192 				}
193 				if (equalsAny(operator, "|", "||", "or")) {
194 					if (!filter) {
195 						continue;
196 					}
197 					else {
198 						return filter;
199 					}
200 				}
201 			}
202 			else if (jsonLever.isJsonObject(condition)) {
203 				String type = jsonLever.getFieldAsString(condition, "type");
204 				type = null != type && queryTypes.containsKey(type) ? type : "=";
205 				filter = queryTypes.get(type).apply(json, condition.getAsJsonObject());
206 			}
207 		}
208 		return filter;
209 	}

211 	/**
212 	 * Equals filter.
213 	 *
214 	 * @param json the json
215 	 * @param condition the condition
216 	 * @return the boolean
217 	 */
218 	private Boolean equalsFilter(JsonElement json, JsonObject condition) {
219 		return jsonLever.getAsJsonElement(condition.get("value"), JsonNull.INSTANCE).equals(
220 				jsonLever.getField(json, jsonLever.getAsJsonArray(condition.get("field"), new JsonArray())));
221 	}
222 
223 	/**
224 	 * Not equals filter.
225 	 *
226 	 * @param json the json
227 	 * @param condition the condition
228 	 * @return the boolean
229 	 */
230 	private Boolean notEqualsFilter(JsonElement json, JsonObject condition) {
231 		return !equalsFilter(json, condition);
232 	}
233 
234 	/**
235 	 * In filter.
236 	 *
237 	 * @param json the json
238 	 * @param condition the condition
239 	 * @return the boolean
240 	 */
241 	private Boolean inFilter(JsonElement json, JsonObject condition) {
242 		return jsonLever.getAsJsonArray(condition.get("value"), new JsonArray()).contains(
243 				jsonLever.getField(json, jsonLever.getAsJsonArray(condition.get("field"), new JsonArray())));
244 	}
245 
246 	/**
247 	 * Notin filter.
248 	 *
249 	 * @param json the json
250 	 * @param condition the condition
251 	 * @return the boolean
252 	 */
253 	private Boolean notinFilter(JsonElement json, JsonObject condition) {
254 		return !inFilter(json, condition);
255 	}
256 
257 	/**
258 	 * Equals any.
259 	 *
260 	 * @param str the str
261 	 * @param matches the matches
262 	 * @return the boolean
263 	 */
264 	private Boolean equalsAny(String str, String... matches) {
265 		Boolean equals = Boolean.TRUE;
266 		for (String string : matches) {
267 			equals = equals && StringUtils.equals(str, string);
268 			if (!equals) {
269 				break;
270 			}
271 		}
272 		return equals;
273 	}
274 
275 	/**
276 	 * Validate.
277 	 *
278 	 * @param query the query
279 	 * @return the string
280 	 */
281 	private String validate(JsonObject query) {
282 		if (jsonLever.isNull(jsonLever.getField(query, "select"))) {
283 			return "select field of the query cannot be null";
284 		}
285 		JsonElement from = query.get("from");
286 		if (!(jsonLever.isJsonObject(from) || jsonLever.isJsonArray(from))) {
287 			return "from field must be a valid json object or array";
288 		}
289 		JsonElement where = jsonLever.getField(query, "where");
290 		if (jsonLever.isNotNull(where) && !jsonLever.isJsonArray(where)) {
291 			return "where field must be a valid json array";
292 		}
293 		JsonElement limit = jsonLever.getField(query, "limit");
294 		if (jsonLever.isNotNull(limit) && !jsonLever.isLong(limit)) {
295 			return "limit must be a valid number";
296 		}
297 		return null;
298 	}
