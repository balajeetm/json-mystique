10  package com.futuresight.util.mystique;
11  
12  import java.util.List;
13  
14  import org.apache.commons.collections.CollectionUtils;
15  import org.apache.commons.lang.StringUtils;
16  import org.springframework.stereotype.Component;
17  
18  import com.futuresight.util.mystique.lever.MysCon;
19  import com.google.gson.JsonElement;
20  import com.google.gson.JsonObject;
21  import com.google.gson.JsonPrimitive;
22  
23  /**
24   * The Class ConcatMystTurn.
25   *
26   * @author balajeetm
27   */
28  @Component
29  public class ConcatMystTurn extends AbstractMystTurn {
30  
31  	/* (non-Javadoc)
32  	 * @see com.futuresight.util.mystique.Mystique#transform(java.util.List, com.google.gson.JsonObject, java.lang.String)
33  	 */
34  	@Override
35  	public JsonElement transmute(List<JsonElement> source, JsonObject deps, JsonObject aces, JsonObject turn) {
36  		StringBuilder stringBuilder = new StringBuilder();
37  		if (CollectionUtils.isNotEmpty(source)) {
38  			String separator = jsonLever.getAsString(turn.get(MysCon.SEPARATOR), MysCon.EMPTY);
39  			for (int count = 0; count < source.size(); count++) {
40  				JsonElement granularSource = getGranularSource(source.get(count), turn, deps, aces);
41  				if (count != 0) {
42  					stringBuilder.append(separator);
43  				}
44  				stringBuilder.append(StringUtils.strip(granularSource.toString(), MysCon.DOUBLE_QUOTES));
45  			}
46  		}
47  		return new JsonPrimitive(stringBuilder.toString());
48  	}
49  
50  }
